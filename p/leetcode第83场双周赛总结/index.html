class Solution {
public:
    string bestHand(vector<int>& ranks, vector<char>& suits) {
        map<char,int> s;
        map<int,int> r;
        for(auto &x: ranks) r[x]++;
        for(auto &x: suits) s[x]++;
        
        if(s.size()==1) return "Flush";
        
        for(auto &x: r){
            if(x.second>=3) return "Three of a Kind";
        }
        
        for(auto &x: r){
            if(x.second==2) return "Pair";
        }
        
        return "High Card";
        
    }
};

#define ll long long
class Solution {
public:
    long long zeroFilledSubarray(vector<int>& nums) {
        int n = nums.size();
        ll res=0;
        
        int i,l=-1,r=0;
        for(i=0;i<n;i++){
           if(nums[i]==0){
               if(l==-1) l=i,r=i;
               else r=i;
           }else{
               if(l!=-1){
                   int len=r-l+1;
                   res += 1ll*len*(len+1)/2;
                   l=-1;
               }
           }
        }
        
        if(nums[n-1]==0){
            int len=n-1-l+1;
            res += 1ll*len*(len+1)/2;
        }
        
        return res;
    }
};

class NumberContainers {
public:
    map<int,set<int>> tab;
    map<int,int> a;
    NumberContainers() {

    }
    
    void change(int index, int number) {
        if(!a.count(index)){
            a[index] = number;
            tab[number].insert(index);
        }else{
            int bef = a[index];
            a[index] = number;
            tab[bef].erase(index);
            tab[number].insert(index);
        }
    }
    
    int find(int number) {
        if(!tab.count(number)) return -1;
        if(tab[number].empty()) return -1;
        return *tab[number].begin();
    }
};

class Solution {
public:
    int shortestSequence(vector<int>& rolls, int k) {
        set<int> st;
        int res=0;
        for(auto &x: rolls){
            st.insert(x);
            if(st.size()==k){
                st.clear();
                res++;
            }
        }
        return res+1;
    }
};
