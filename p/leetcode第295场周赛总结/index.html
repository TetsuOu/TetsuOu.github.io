class Solution {
public:
    int rearrangeCharacters(string s, string target) {
        vector<int> cnt(26,0),t(26,0);
        for(auto &x: s) cnt[x-'a']++;
        for(auto &x: target) t[x-'a']++;
        int res=100;
        for(int i=0;i<26;i++){
            if(!t[i]) continue;
            res = min(res, cnt[i]/t[i]);
        }
        return res;
    }
};

class Solution {
public:
    string discountPrices(string sentence, int discount) {
        stringstream ss(sentence);
        string res,cur;
        while(ss>>cur){
            if(cur.size()>1 && cur[0]=='$'){
                bool flag=true;
                for(int i=1;i<cur.size()&&flag;i++) {
                    if(cur[i]>='0' && cur[i]<='9') continue;
                    else flag=false;
                }
                if(!flag){
                    res += cur+' ';
                    continue;
                }
                long long price = stol(cur.substr(1));
                double p = 1.00*price*(100-discount)/100.00;
                stringstream out;
                out<<fixed<<setprecision(2)<<p;
                string s = out.str();
                cur = '$'+s;
            }
            res += cur+' ';
        }
        res = res.substr(0,res.size()-1);
        return res;
    }
};

class Solution:
    def discountPrices(self, sentence: str, discount: int) -> str:
        arr = []
        for x in sentence.split():
            if x.startswith('$') and x[1:].isdigit():
                num = float(x[1:]) * (100 - float(discount)) / 100
                arr.append('${:.2f}'.format(num))
            else:
                arr.append(x)
        return ' '.join(arr)

const int MAXN = 1e5+5;
#define ls(p) (p<<1)
#define rs(p) (p<<1|1)
class Solution {
public:
    struct node{
        int maxv;
    }T[MAXN<<2];
    void push_up(int p){
        T[p].maxv=max(T[ls(p)].maxv,T[rs(p)].maxv);
    }
    void build(int p, int l, int r){
        T[p]={0};
        if(l==r) return;
        int mid=(l+r)>>1;
        build(ls(p),l,mid);
        build(rs(p),mid+1,r);
    }
    void update(int p, int l, int r, int nr, int k){
        if(l==r && l==nr){
            T[p].maxv = k;
            return;
        }
        int mid=(l+r)>>1;
        if(nr<=mid) update(ls(p),l,mid,nr,k);
        else update(rs(p),mid+1,r,nr,k);
        push_up(p);
    }
    int query(int p, int l, int r, int nl, int nr){
        if(nl<=l&&nr>=r) return T[p].maxv;
        int mid=(l+r)>>1;
        int res=0;
        if(nl<=mid) res=max(res,query(ls(p),l,mid,nl,nr));
        if(nr>mid) res=max(res,query(rs(p),mid+1,r,nl,nr));
        return res;
    }
    int totalSteps(vector<int>& nums) {
        int n = nums.size();
        vector<int> L(n,-1);//L[i] å·¦è¾¹ç¬¬ä¸€ä¸ªæ¯”nums[i]å¤§çš„æ•°çš„ä½ç½®
        stack<int> stk;
        stk.push(0);
        for(int i=1;i<n;i++){
            while(!stk.empty() && nums[i]>=nums[stk.top()]) stk.pop();
            if(stk.empty()) L[i] = -1;
            else L[i] = stk.top();
            stk.push(i);
        }

        vector<int> t(n,0);//t[i] åˆ é™¤nums[i]æ‰€éœ€æ‰§è¡Œçš„æ“ä½œæ•°
        int res = 0;
        
        build(1,1,n);
        for(int i=0;i<n;i++){
            if(L[i]==-1) t[i]=0;//nums[i]å·¦è¾¹æ²¡æœ‰æ¯”å®ƒæ›´å¤§çš„æ•°ï¼Œä¸ä¼šè¢«åˆ é™¤
            //else if(L[i]==i-1) t[i]=1;//ä¼šè¢«ç›¸é‚»å·¦è¾¹å…ƒç´ åˆ æ‰ï¼Œæ­¥æ•°ä¸º1
            else{
                //t[i]=max(t[l+1..i-1])
                t[i]=query(1,1,n,L[i]+1+1,i-1+1)+1;
            }
            update(1,1,n,i+1,t[i]);
            res = max(res,t[i]);
        }

        return res;
    }
};

    };

        vector<vector<int>> vis(n,vector<int>(m,0));
        
        priority_queue<node> que;
        que.emplace(0,0,0);
        dst[0][0]=0;
        vis[0][0]=1;
        function<bool(int,int)> isok = [&](int p,int q){
            if(p<0||p>=n) return false;
            if(q<0||q>=m) return false;
            return true;
        };   
        while(!que.empty()){
            auto [x,y,del] = que.top();

            if(x==n-1&&y==m-1) return del;

                int dx=x+dir[i][0];
                int dy=y+dir[i][1];
                if(!isok(dx,dy)) continue;
                if(vis[dx][dy]) continue;
                if(del+grid[dx][dy]<dst[dx][dy]){
                    dst[dx][dy]=del+grid[dx][dy];
                    que.push(node(dx,dy,dst[dx][dy]));

            }
            
        }

    }

class Solution {
public:
    struct node{
        int x,y;
        int del;
        node(int _x,int _y,int _d){
            x=_x,y=_y,del=_d;
        }
    };

        que.emplace_back(0,0,0);
