const int INF = 0x3f3f3f3f;
class Solution {
public:
    int coinChange(vector<int>& coins, int amount) {
        int n = coins.size();
        vector<vector<int>> dp(n,vector<int>(amount+1,INF));
        //dp[i][j]å‰iç§ç¡¬å¸å‡‘æˆjæ‰€éœ€çš„æœ€å°‘ç¡¬å¸ä¸ªæ•°
        for(int j=0;j<=amount;j++){
            if(j%coins[0]==0) dp[0][j]=j/coins[0];
        }

        for(int i=1;i<n;i++){
            for(int j=0;j<=amount;j++){
                dp[i][j] = min(dp[i][j],dp[i-1][j]);
                if(j>=coins[i]) dp[i][j]=min(dp[i][j],dp[i][j-coins[i]]+1);
            }
        }

        if(dp[n-1][amount]==INF) return -1;
        else return dp[n-1][amount];
    }
};

const int INF = 0x3f3f3f3f;
class Solution {
public:
    int coinChange(vector<int>& coins, int amount) {
        int n = coins.size();
        vector<int> dp(amount+1,INF);
        //dp[j] å‡‘æˆjæ‰€éœ€çš„æœ€å°‘ç¡¬å¸ä¸ªæ•°

class Solution {
public:
    int combinationSum4(vector<int>& nums, int target) {
        int len=target;
        vector<vector<ull>> dp(len+1,vector<ull>(target+1,0));

                for(auto &x:nums){

    int numSquares(int n) {
        vector<int> v;//å®¹é‡æ•°ç»„
        for(int i=1;i*i<=n;i++){

        int len=v.size();//lenä»¶ç‰©å“

    bool wordBreak(string s, vector<string>& wordDict) {
        unordered_set<string> tab(wordDict.begin(),wordDict.end());
        int n = s.size();
        vector<bool> dp(n+1,false);

            }
        }
        return dp[n];
    }
};
