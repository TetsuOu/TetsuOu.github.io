class Solution {
public:
    int countAsterisks(string s) {
        int n = s.size();
        int in = 0, all = 0;
        bool hasleft=false;
        for(auto &ch: s){
            if(ch=='*') all++;
            if(ch=='|'){
                if(hasleft) hasleft=false;
                else{
                    hasleft=true;
                }
            }
            if(hasleft&&ch=='*') in++;
        }
        return all-in;
    }
};

const int MAXN=1e5+5;
int p[MAXN],s[MAXN];
int setfind(int x){
    if(p[x]!=x) p[x]=setfind(p[x]);
    return p[x];
}
void setunion(int x, int y){
    if(setfind(x)==setfind(y)) return;
    s[setfind(y)]+=s[setfind(x)];
    p[setfind(x)]=setfind(y);
    return;
}
int setsize(int x){
    return s[setfind(x)];
}
class Solution {
public:
    void init(int n){
        for(int i=0;i<n;i++){
            p[i]=i;
            s[i]=1;
        }
    }
    long long countPairs(int n, vector<vector<int>>& edges) {
        long long res = 0;
        init(n);
        for(auto &x: edges){
            setunion(x[0],x[1]);
        }
        for(int i=0;i<n;i++){
            int sz = s[setfind(i)];
            res += n-sz;
        }
        return res/2;
    }
};

class Solution {
    vector<vector<int>> e;
    vector<bool> vis;

    int bfs(int S) {
        int ret = 0;
        queue<int> q;
        q.push(S); vis[S] = true;
        while (!q.empty()) {
            ret++;
            int sn = q.front(); q.pop();
            for (int fn : e[sn]) if (!vis[fn]) q.push(fn), vis[fn] = true;
        }
        return ret;
    }

public:
    long long countPairs(int n, vector<vector<int>>& edges) {
        e.resize(n, vector<int>());
        vis.resize(n, false);
        for (auto &edge : edges) e[edge[0]].push_back(edge[1]), e[edge[1]].push_back(edge[0]);

        long long ans = 0;
        for (int i = 0; i < n; i++) if (!vis[i]) {
            int t = bfs(i);
            ans += 1LL * t * (n - t);
        }
        return ans / 2;
    }
};

class Solution {
public:
    int maximumXOR(vector<int>& nums) {
        int n = nums.size();
        vector<int> cnt(32,0);
        for(auto &num: nums){
            for(int i=0;i<32;i++){
                if((num>>i)&1) cnt[i]++;
            }
        }
        int res = 0;
        for(int i=0;i<32;i++){
            if(cnt[i]==0) continue;
            res += (1<<i);
        }
        return res;
    }
};

                    for(int p=1;p<=6;p++){//æšä¸¾ç¬¬i-2ä½çš„æ•°
                        if(__gcd(p,j)!=1) continue;
                        if(p==k) continue;
                        dp[i][j][k]=(dp[i][j][k]+dp[i-1][p][j])%MOD;
                    }
                }
            }
        }
        ll res = 0;
        for(int j=1;j<=6;j++){
            for(int k=1;k<=6;k++){
                res = (res+dp[n][j][k])%MOD;
            }
        }
        
        return res;
    }
