            maxprofit = max(maxprofit, price - minprice);

class Solution {
public:
    int maxProfit(int k, vector<int>& prices) {
        int n = prices.size();
        if(k==0||n==0) return 0;
        int dp[n][2][k+1];
        //dp[i][0][p]  ÂâçiÂ§©Ôºå‰∏îÁ¨¨iÂ§©Êú™ÊåÅÊúâËÇ°Á•®ÔºåÂ∑≤Áªè‰∫§Êòì‰∫ÜpÊ¨°

            dp[0][1][p] = -prices[0];

        for(int i=1;i<n;i++){
            dp[i][0][0] = 0;
            dp[i][1][0] = max(dp[i-1][0][0]-prices[i], dp[i-1][1][0]);

            }
        }

        int res=0;
        for(int p=0;p<=k;p++){
            res=max(res,dp[n-1][0][p]);
        }
        return res;
    }
